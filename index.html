Mup Lab 7 tuesday 2-4

Q:Write an ALP that takes in user key press one by one until enter key (ODH) is pressed. The characters entered by the user must be stored in a file zyx.txt. The file is not already present; it has to be created. You can assume that user will not enter more than 100 characters (this is only for reserving storage space)

Solution:

.model tiny

.486

.data

max1 db 100

act1 db ?

inp1 db 100 dup('$')

fname db 'hex.txt',0

handle dw ?

.code

.startup

;enter text

lea dx,max1

mov ah,0ah

int 21h

;create file

mov ah,3ch

lea dx,fname

mov cl,2

int 21h

mov handle,ax

lea si,act1

mov al,[si]

and al,0fh

;write to the file

mov ah,40h

mov bx,handle

movzx cx,al

lea dx,inp1

int 21h

;close the file

mov ah,3eh

mov bx, handle

int 21h

.exit

end
