Mup Lab 7 tuesday 4-6

Q:Write an ALP that r eads the first character of the file abc.txt ( available on the site itself). If the character is in caps it converts it to small letters and displays, it. If the character is a small lette r it converts it to caps and displays, it. If the character read is a special character or numeral it just displays, it.

Solution:

.model tiny

.data

fname db 'abc.txt',0

handle dw ?

dat1 db 5 dup('$')

.code

.startup

;open the file

mov ah,3dh

mov al,0

lea dx,fname

int 21h

mov handle,ax

;read it

mov ah,3fh

mov bx,handle

mov cx,1

lea dx,dat1

int 21h

;check

lea si,dat1

mov bl,[si]

cmp bl,5ah

jle x1

;small to caps

sub bl,20h

cmp ax,ax

je x2

x1: cmp bl,39h

jle x2

;caps to small

add bl,20h

;display

x2: mov dl,bl

mov ah,02h

int 21h

;close the file

mov ah,3eh

mov bx, handle

int 21h

.exit

end

Mup Lab 7 tuesday 2-4

Q:Write an ALP that takes in user key press one by one until enter key (ODH) is pressed. The characters entered by the user must be stored in a file zyx.txt. The file is not already present; it has to be created. You can assume that user will not enter more than 100 characters (this is only for reserving storage space)

Solution:

.model tiny

.486

.data

max1 db 100

act1 db ?

inp1 db 100 dup('$')

fname db 'hex.txt',0

handle dw ?

.code

.startup

;enter text

lea dx,max1

mov ah,0ah

int 21h

;create file

mov ah,3ch

lea dx,fname

mov cl,2

int 21h

mov handle,ax

lea si,act1

mov al,[si]

and al,0fh

;write to the file

mov ah,40h

mov bx,handle

movzx cx,al

lea dx,inp1

int 21h

;close the file

mov ah,3eh

mov bx, handle

int 21h

.exit

end
